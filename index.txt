                document.getElementById('resetGameButton').disabled = false;
            }
        }
        
        /** Renders the current game state to the UI. */
        function renderUI() {
            updateMainDisplay();

            // Tab visibility logic
            const gameContent = document.getElementById('gameTabContent');
            const rulesContent = document.getElementById('rulesTabContent');
            const gameTabBtn = document.getElementById('gameTabBtn');
            const rulesTabBtn = document.getElementById('rulesTabBtn');

            if (gameState.activeTab === 'game') {
                gameContent.classList.remove('hidden');
                rulesContent.classList.add('hidden');
                
                gameTabBtn.classList.add('bg-indigo-50/50', 'border-indigo-600', 'text-indigo-700');
                gameTabBtn.classList.remove('text-gray-600', 'hover:bg-gray-50/50', 'border-transparent');
                
                rulesTabBtn.classList.remove('bg-indigo-50/50', 'border-indigo-600', 'text-indigo-700');
                rulesTabBtn.classList.add('text-gray-600', 'hover:bg-gray-50/50', 'border-transparent');
            } else { // activeTab === 'rules'
                gameContent.classList.add('hidden');
                rulesContent.classList.remove('hidden');

                rulesTabBtn.classList.add('bg-indigo-50/50', 'border-indigo-600', 'text-indigo-700');
                rulesTabBtn.classList.remove('text-gray-600', 'hover:bg-gray-50/50', 'border-transparent');

                gameTabBtn.classList.remove('bg-indigo-50/50', 'border-indigo-600', 'text-indigo-700');
                gameTabBtn.classList.add('text-gray-600', 'hover:bg-gray-50/50', 'border-transparent');
            }
        }
        
        window.switchTab = function(newTab) {
            if (gameState.activeTab !== newTab) {
                gameState.activeTab = newTab;
                renderUI(); 
                saveGameState(); // Save tab state
            }
        }

        // --- Firestore Operations ---

        /** Gets the document reference for the current user's session. */
        function getSessionDocRef() {
            const sessionPath = `/artifacts/${appId}/users/${userId}/catch41_sessions`;
            return doc(db, sessionPath, 'session');
        }

        /** Loads the game state from Firestore and sets up the listener. */
        function loadGameState() {
            if (!isAuthReady || !db || !userId) return;

            const docRef = getSessionDocRef();
            
            onSnapshot(docRef, (docSnap) => {
                if (docSnap.exists()) {
                    const loadedState = docSnap.data();
                    gameState = {
                        ...loadedState,
                        currentOut: parseInt(loadedState.currentOut),
                        previousState: loadedState.previousState || null,
                        activeTab: loadedState.activeTab || 'game'
                    };
                } else {
                    if (gameState.currentOut === MIN_OUT) {
                        saveGameState();
                    }
                }
                renderUI(); 
            }, (error) => {
                console.error("Error setting up onSnapshot listener:", error);
            });
        }
        
        /** Saves the current game state to Firestore. */
        async function saveGameState() {
            if (!isAuthReady || !db || !userId) {
                return;
            }
            
            gameState.lastUpdated = new Date().toISOString();
            const docRef = getSessionDocRef();
            
            const stateToSave = {
                totalScore: gameState.totalScore,
                currentOut: gameState.currentOut,
                outsCompleted: gameState.outsCompleted,
                totalDartsThrown: gameState.totalDartsThrown,
                outsData: gameState.outsData,
                previousState: gameState.previousState,
                lastUpdated: gameState.lastUpdated,
                activeTab: gameState.activeTab
            };

            try {
                await setDoc(docRef, stateToSave, { merge: true });
            } catch (e) {
                console.error("Error writing document: ", e);
            }
        }
        
        // --- Game Logic ---

        /** Handles the user recording a finish. */
        async function recordFinish(dartsInput) {
            const errorElement = document.getElementById('errorMessage');
            
            const dartsUsed = dartsInput === 0 ? 7 : dartsInput;

            if (gameState.currentOut > MAX_OUT) {
                errorElement.textContent = "Game finished! Please reset to start a new session.";
                return;
            }

            errorElement.textContent = "";
            
            // 0. Save current state for potential Undo
            const stateToSave = JSON.parse(JSON.stringify(gameState));
            delete stateToSave.previousState;
            gameState.previousState = stateToSave;
            
            const currentOut = gameState.currentOut;
            const points = calculatePoints(dartsUsed, currentOut); 
            
            // 1. Update Game State
            if (!gameState.outsData[currentOut]) {
                gameState.outsCompleted += 1;
            } else {
                // If re-recording the same out, subtract old score/darts first
                gameState.totalScore -= (gameState.outsData[currentOut].score || 0);
                gameState.totalDartsThrown -= (gameState.outsData[currentOut].darts || 0);
            }

            gameState.totalScore += points;
            gameState.totalDartsThrown += dartsUsed; 
            gameState.outsData[currentOut] = { darts: dartsUsed, score: points };
            
            // 2. Move to next out
            if (gameState.currentOut <= MAX_OUT) {
                gameState.currentOut += 1;
            }

            // Cap score at MAX_SCORE
            if (gameState.totalScore > MAX_SCORE) {
                gameState.totalScore = MAX_SCORE;
            }

            updateMainDisplay(); // Instant update
            await saveGameState(); // Save asynchronously
        }
        
        /** Reverts the game state to the point before the last successful recordFinish. */
        async function undoLastScore() {
            if (!gameState.previousState) {
                console.warn("No previous state available to undo.");
                return;
            }
            
            document.getElementById('errorMessage').textContent = "";

            const previousState = gameState.previousState;
            const currentTab = gameState.activeTab;

            gameState = { 
                ...previousState, 
                previousState: null,
                activeTab: currentTab
            }; 
            
            updateMainDisplay(); 
            await saveGameState(); 
        }
        
        /** Resets the game state and saves the new session. */
        async function resetSession() {
            gameState = {
                totalScore: 0,
                currentOut: MIN_OUT,
                outsCompleted: 0,
                totalDartsThrown: 0,
                outsData: {},
                lastUpdated: new Date().toISOString(),
                previousState: null,
                activeTab: 'game' // Always reset to the game tab
            };
            
            updateMainDisplay(); 
            await saveGameState();
            renderUI();
        }

        // --- Download Functionality ---
        
        /** Downloads the current HTML file for offline use */
        function downloadHTML() {
            // Get the current HTML content
            const htmlContent = document.documentElement.outerHTML;
            
            // Create a blob with the HTML content
            const blob = new Blob(['<!DOCTYPE html>\n' + htmlContent], { type: 'text/html' });
            
            // Create a download link
            const downloadLink = document.createElement('a');
            downloadLink.href = URL.createObjectURL(blob);
            downloadLink.download = 'catch41-practice.html';
            
            // Trigger the download
            document.body.appendChild(downloadLink);
            downloadLink.click();
            document.body.removeChild(downloadLink);
            
            // Clean up the object URL
            URL.revokeObjectURL(downloadLink.href);
        }

        // --- Initialization ---

        /** Initializes Firebase and sets up authentication. */
        async function initializeFirebase() {
            try {
                if (Object.keys(firebaseConfig).length === 0) {
                    console.error('Firebase initialization skipped: Configuration object is empty. Data saving is disabled.');
                    isAuthReady = true; 
                    return;
                }
                
                app = initializeApp(firebaseConfig);
                db = getFirestore(app);
                auth = getAuth(app);
                
                if (initialAuthToken) {
                    await signInWithCustomToken(auth, initialAuthToken);
                } else {
                    await signInAnonymously(auth);
                }
                
                onAuthStateChanged(auth, (user) => {
                    if (user) {
                        userId = user.uid || crypto.randomUUID(); 
                        isAuthReady = true;
                        loadGameState();
                    } else {
                        userId = crypto.randomUUID();
                        isAuthReady = true;
                        loadGameState();
                    }
                });

            } catch (e) {
                console.error("Firebase initialization failed:", e);
            }
        }
        
        // --- Event Listeners ---
        window.onload = function() {
            initializeFirebase();
            
            // 1. Download Button Listener
            document.getElementById('downloadButton').addEventListener('click', downloadHTML);
            
            // 2. Completion Modal Reset Button Listener (Direct Reset)
            document.getElementById('completionModalResetButton').addEventListener('click', resetSession);
            
            // 3. In-Game Reset Button Listener (Opens Confirmation Modal)
            document.getElementById('resetGameButton').addEventListener('click', showConfirmResetModal);

            // 4. New Confirmation Modal Button Listeners
            document.getElementById('confirmResetConfirmButton').addEventListener('click', async () => {
                hideConfirmResetModal();
                await resetSession();
            });
            document.getElementById('confirmResetCancelButton').addEventListener('click', hideConfirmResetModal);

            // 5. Undo Button Listener
            document.getElementById('undoButton').addEventListener('click', undoLastScore);
            
            // 6. Combined 456 Button Logic (no dropdown)
            // Button now just submits 4 darts
            // Keyboard input for 5/6 still works
            
            // 7. Hotkey (Keyboard Listener)
            window.addEventListener('keydown', (e) => {
                const key = e.key;
                const isGameActive = gameState.activeTab === 'game' &&
                                     !document.getElementById('completionModal').classList.contains('flex') &&
                                     !document.getElementById('confirmResetModal').classList.contains('flex'); // Prevent input if modal is open

                if (isGameActive) {
                    const allowedScoreKeys = ['0', '2', '3', '4', '5', '6']; 
                    
                    if (allowedScoreKeys.includes(key)) {
                        e.preventDefault(); 
                        recordFinish(parseInt(key));
                        dropdown.classList.add('hidden'); // Close dropdown on keyboard input
                    } else if (key === 'Backspace') {
                        const undoButton = document.getElementById('undoButton');
                        if (!undoButton.disabled) {
                            e.preventDefault(); 
                            undoLastScore();
                        }
                    }
                }
            });

            // 8. On-screen Button Listeners
            document.querySelectorAll('[data-darts]').forEach(button => {
                button.addEventListener('click', () => {
                    const darts = parseInt(button.getAttribute('data-darts'));
                    recordFinish(darts);
                    dropdown.classList.add('hidden'); // Close dropdown after selection
                });
            });
        };
